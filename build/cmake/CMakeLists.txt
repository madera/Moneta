CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(moneta)
SET(PROJECT_ROOT "${moneta_SOURCE_DIR}/../..")

SET(MONETA_INCLUDE "${PROJECT_ROOT}/moneta")
INCLUDE_DIRECTORIES("${MONETA_INCLUDE}")

SET(TEST_ROOT "${PROJECT_ROOT}/test")
FILE(GLOB_RECURSE TEST_SOURCE_FILES "${TEST_ROOT}/*.cxx")

MESSAGE(STATUS "PROJECT_ROOT     : " ${PROJECT_ROOT}     )
MESSAGE(STATUS "MONETA_INCLUDE   : " ${MONETA_INCLUDE}   )
MESSAGE(STATUS "TEST_ROOT        : " ${TEST_ROOT}        )
MESSAGE(STATUS "TEST_SOURCE_FILES: " ${TEST_SOURCE_FILES})

ADD_EXECUTABLE(test ${TEST_SOURCE_FILES})
